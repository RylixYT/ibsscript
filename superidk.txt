$csvPath = "Pfad\\zum\\Verzeichnis\\mit\\CSV-Dateien"
$outputFilePath = "Ziel\\und\\Name\\der\\Zusammenfassung.csv"


$csvContent = @()


$csvFiles = Get-ChildItem -Path $csvPath -Filter *.csv


foreach ($file in $csvFiles) {
    # Lese den Inhalt der CSV-Datei
    $content = Get-Content -Path $file.FullName
    # Füge den Inhalt zur $csvContent-Array hinzu
    $csvContent += $content
}


$csvContent | Out-File -FilePath $outputFilePath -Encoding UTF8

Write-Host "Die CSV-Dateien wurden erfolgreich zu $outputFilePath zusammengeführt."
