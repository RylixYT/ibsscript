$computerName = $env:COMPUTERNAME
$outputFileName = "lastUser_Serial_$computerName.csv"

$systemInfo = Get-WmiObject -Class Win32_ComputerSystem
$serialNumber = (Get-WmiObject -Class Win32_BIOS).SerialNumber
$lastUser = $systemInfo.UserName

if ($lastUser) {
    $lastUser = $lastUser.Split('\')[-1]
    $outputData = [PSCustomObject]@{
        "ComputerName" = $computerName
        "SerialNumber" = $serialNumber
        "LastUser" = $lastUser
    }
    $outputData | Export-Csv -Path $outputFileName -NoTypeInformation
    Write-Output "Die Informationen wurden in die CSV-Datei '$outputFileName' exportiert."
} else {
    Write-Output "Es wurde kein Benutzer auf $computerName gefunden."
}
